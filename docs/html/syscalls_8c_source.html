88 caddr_t _sbrk(int incr)
89 {
90  extern char end asm("end");
91  static char *heap_end;
92  char *prev_heap_end;
93 
94  if (heap_end == 0)
95  heap_end = &end;
96 
97  prev_heap_end = heap_end;
98  if (heap_end + incr > stack_ptr)
99  {
100 // write(1, "Heap and stack collision\n", 25);
101 // abort();
102  errno = ENOMEM;
103  return (caddr_t) -1;
104  }
105 
106  heap_end += incr;
107 
108  return (caddr_t) prev_heap_end;
109 }
